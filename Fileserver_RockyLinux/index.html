vim /etc/samba/smb.conf

    [global]
    workgroup = ESHARKNET
    #hosts allow = 127. 192.168.0
    server string = Servidor de Arquivos %v
    netbios name = servidor
    security = user
    map to guest = bad user
    dns proxy = no
    ntlm auth = true

    passdb backend = tdbsam

    printing = cups
    printcap name = cups
    load printers = yes
    cups options = raw

[home]
    comment = Diretorio Home
    valid users = %S, %D%w%S
    browseable = No
    read only = No
    inherit acls = Yes

[printers]
    comment = All Printers
    path = /var/tmp
    printable = Yes
    create mask = 0600
    browseable = No

[print$]
    comment = Printer Drivers
    path = /var/lib/samba/drivers
    write list = @printadmin root
    force group = @printadmin
    create mask = 0664
    directory mask = 0775

[ARQUIVOS]
    comment = Compartilhamentos de Rede
    path = /srv/samba/ARQUIVOS
    browseable = Yes
    writeable = Yes
    guest ok = Yes

[Diretoria]
    path = /srv/samba/Diretoria
    valid users = @diretoria
    guest ok = no
    writable = no
    browsable = yes

[Financeiro]
    path = /srv/samba/Financeiro

    guest ok = no

    browsable = yes
